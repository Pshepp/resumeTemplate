% job listing environment
\newenvironment{job}[5]
{%
	\vspace{2pt}
	\begin{tabularx}{\linewidth}{@{}l Y r@{}}
		\textbf{#1 \textbar} #2 & \hfill &  #3 \\
		\textit{#4} & \hfill &  #5\\
	\end{tabularx}
	\begin{itemize}
}
{%
	\end{itemize}
}

% personal project enviroment
\newenvironment{project}[1]
{%
	\vspace{2pt}
	\textbf{#1}\par
	\begin{itemize}
} 
{%
	\end{itemize}
}

%extracurriculars and awards stuff, one is for allowing a bullet list within
\newenvironment{extraList}[2]
{%
	\begin{tabularx}{\linewidth}{@{}l Y r@{}}
		#1 & \hfill & #2 \\
	\end{tabularx}
	\begin{itemize}
}
{%
	\end{itemize}
}

\newenvironment{extra}[2]
{%
	\begin{tabularx}{\linewidth}{@{}l Y r@{}}
		#1 & \hfill & #2 \\
	\end{tabularx}
}
{}

%only using commands because nothing in body of these
\newcommand{\education}[5]
{%
	\begin{tabularx}{\linewidth}{@{}l Y r@{}}
		\textbf{#1 \textbar} #2 & \hfill & Graduation: #3 \\
		\textit{#4} & \hfill & GPA #5/4.0
	\end{tabularx}%
}

\newcommand{\publication}[4]
{%
	\textbf{#1}\par
	\begin{tabularx}{\linewidth}{@{}X r@{}}	
		#2 & #3: \href{#4}{#4} \\
	\end{tabularx}%
}
